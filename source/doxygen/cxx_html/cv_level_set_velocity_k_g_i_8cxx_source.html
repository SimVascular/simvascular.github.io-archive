97 // -------------------
98 // SetUpPotentialField
99 // -------------------
100 // To be called only from one of the SetImage methods. The basic idea
101 // here is that the image-dependent potential field of the geodesic
102 // approach needs to define potential wells at edges. This field is
103 // essentially only a function of the image itself. As Caselles
104 // notes, the field function g should be a decreasing function of
105 // gradient. Malladi and Sethian use g = - | grad(I) |. In this
106 // method, we will set up a second "image" which contains the values
107 // of the potential field. This field's gradient can then be computed
108 // and used in the solution of the geodesic approach.
109 
110 void cvLevelSetVelocityKGI::SetUpPotentialField( Image_T *img )
111 {
112  Image_T *tmp;
113  int numPix, i;
114  double mag;
115  double *data;
116 
117  if ( potential_ != NULL ) {
118  Image_Delete( potential_ );
119  }
120  ComputeImageGrad( img );
121 
122  numPix = img->imgDims[0] * img->imgDims[1] * img->imgDims[2];
123  data = new double [numPix];
124  for ( i = 0; i < numPix; i++ ) {
125  mag = Magnitude( img->pixels[i].gradX,
126  img->pixels[i].gradY,
127  img->pixels[i].gradZ );
128  // data[i] = 1.0 / ( 1.0 + mag );
129  data[i] = - mag;
130  }
131  tmp = CreateImage( (void*)data, numPix, "-double", img->imgDims,
132  img->pixelDims );
133  delete [] data;
134  if ( tmp == NULL ) {
135  printf("ERR: SetUpPotentialField failed.\n");
136  return;
137  }
138  ComputeImageGrad( tmp );
139  potential_ = tmp;
140  return;
141 }

386 int cvLevelSetVelocityKGI::Evaluate( double pos[], double *f0, double *f1, double v[],
387  int *forceFlag, double toDot[] )
388 {
389  double K, normalGrad;
390  double gradI[3];
391  double n[3];
392  double den, mag;
393  double tmpF0, tmpF1;
394  double signK;
395  double gradg[3];
396 
397  if ( !Valid() ) return CV_ERROR;
398 
399  if ( ls_->GetGrid()->InterpK( pos, &K ) != CV_OK ) return CV_ERROR;
400  if ( ls_->GetGrid()->InterpN( pos, n ) != CV_OK ) return CV_ERROR;
401 
402  if ( GetImageClosed( image_ ) && InBorder( image_, pos, 1 ) ) {
403  toDot[0] = 0.0;
404  toDot[1] = 0.0;
405  toDot[2] = 0.0;
406  *f0 = 0.0;
407  *f1 = 0.0;
408  if ( v != NULL ) {
409  v[0] = 0.0;
410  v[1] = 0.0;
411  v[2] = 0.0;
412  }
413  return CV_OK;
414  }
415 
416  // Look up intensity gradient:
417  if ( ! GetGradIx( image_, pos, &(gradI[0]) ) ) {
418  return CV_ERROR;
419  }
420  if ( ! GetGradIy( image_, pos, &(gradI[1]) ) ) {
421  return CV_ERROR;
422  }
423  if ( image_->dim == 3 ) {
424  if ( ! GetGradIz( image_, pos, &(gradI[2]) ) ) {
425  return CV_ERROR;
426  }
427  } else {
428  gradI[2] = 0.0;
429  }
430 
431  // Look up potential gradient:
432  if ( ! GetGradIx( potential_, pos, &(gradg[0]) ) ) {
433  return CV_ERROR;
434  }
435  if ( ! GetGradIy( potential_, pos, &(gradg[1]) ) ) {
436  return CV_ERROR;
437  }
438  if ( image_->dim == 3 ) {
439  if ( ! GetGradIz( potential_, pos, &(gradg[2]) ) ) {
440  return CV_ERROR;
441  }
442  } else {
443  gradg[2] = 0.0;
444  }
445  toDot[0] = beta_ * gradg[0];
446  toDot[1] = beta_ * gradg[1];
447  toDot[2] = beta_ * gradg[2];
448 
449  normalGrad = Dot( gradI[0], gradI[1], gradI[2], n[0], n[1], n[2] );
450  normalGrad = pow( fabs( normalGrad ), gradIPow_ );
451 
452  den = 1.0 + ( eI_ * normalGrad );
453  tmpF0 = balloonF_ / den;
454  tmpF1 = - eK_ * K / den;
455  mag = tmpF0 + tmpF1;
456 
457  (*forceFlag) = 0;
458 
459  if ( ( localStop_ ) && ( fabs(mag) < stopV_ ) ) {
460 
461  if ( ( fabs(eK_) > ktol_ ) && ( fabs(K) >= fabs(1 / eK_) ) ) {
462 
463  // Curvature K is above specified tolerance (1/eK_). So, now we
464  // want to compare the computed velocity (i.e. mag) with the
465  // stopping criterion stopV_. If ( mag >= stopV_ ), then we
466  // want to just apply mag as it's been computed. Otherwise, we
467  // now want to force a nominal smoothing velocity, even though
468  // computed mag was small.
469 
470  // We have already established by the test of eK_ and the
471  // comparison of K with 1/eK_ that K is non-zero, so sign must
472  // be either positive or negative:
473  if ( fabs( K ) > K ) {
474  signK = -1.0;
475  } else {
476  signK = 1.0;
477  }
478  tmpF0 = - signK * stopV_;
479  tmpF1 = 0.0;
480  (*forceFlag) = 1;
481 
482  } else {
483  tmpF0 = 0.0;
484  tmpF1 = 0.0;
485  }
486  }
487 
488  *f0 = tmpF0;
489  *f1 = tmpF1;
490  mag = (*f0) + (*f1);
491 
492  if ( v != NULL ) {
493  v[0] = n[0] * mag;
494  v[1] = n[1] * mag;
495  v[2] = n[2] * mag;
496  }
497 
498  return CV_OK;
499 }
