63 // list.cc
64 // Routines to manage a singly linked list of "things".
65 // Lists are implemented as templates so that we can store
66 // anything on the list in a type-safe manner.
67 //
68 // A "cvLispListElement" is allocated for each item to be put on the
69 // list; it is de-allocated when the item is removed. This means
70 // we don't need to keep a "next" pointer in every object we
71 // want to put on a list.
72 //
73 // NOTE: Mutual exclusion must be provided by the caller.
74 // If you want a synchronized list, you must use the routines
75 // in synchlist.cc.
76 //
77 // Copyright (c) 1992-1996 The Regents of the University of California.
78 // All rights reserved. See copyright.h for copyright notice and limitation
79 // of liability and disclaimer of warranty provisions.

121 //----------------------------------------------------------------------
122 // cvLispList<T>::Append
123 // Append an "item" to the end of the list.
124 //
125 // Allocate a cvLispListElement to keep track of the item.
126 // If the list is empty, then this will be the only element.
127 // Otherwise, put it at the end.
128 //
129 // "item" is the thing to put on the list.
130 //----------------------------------------------------------------------
131 
132 template <class T>
133 void
134 cvLispList<T>::Append(T item)
135 {
136  cvLispListElement<T> *element = new cvLispListElement<T>(item);
137 
138  Assert(!IsInList(item));
139  if (IsEmpty()) { // list is empty
140  first = element;
141  last = element;
142  } else { // else put it after last
143  last->next = element;
144  last = element;
145  }
146  numInList++;
147  Assert(IsInList(item));
148 }

173 //----------------------------------------------------------------------
174 // cvLispList<T>::RemoveFront
175 // Remove the first "item" from the front of the list.
176 // List must not be empty.
177 //
178 // Returns:
179 // The removed item.
180 //----------------------------------------------------------------------
181 
182 template <class T>
183 T
184 cvLispList<T>::RemoveFront()
185 {
186  cvLispListElement<T> *element = first;
187  T thing;
188 
189  Assert(!IsEmpty());
190 
191  thing = first->item;
192  if (first == last) { // list had one item, now has none
193  first = NULL;
194  last = NULL;
195  } else {
196  first = element->next;
197  }
198  numInList--;
199  delete element;
200  return thing;
201 }

203 //----------------------------------------------------------------------
204 // cvLispList<T>::Remove
205 // Remove a specific item from the list. Must be in the list!
206 //----------------------------------------------------------------------
207 
208 template <class T>
209 void
210 cvLispList<T>::Remove(T item)
211 {
212  cvLispListElement<T> *prev, *ptr;
213  T removed;
214 
215  Assert(IsInList(item));
216 
217  // if first item on list is match, then remove from front
218  if (item == first->item) {
219  removed = RemoveFront();
220  Assert(item == removed);
221  } else {
222  prev = first;
223  for (ptr = first->next; ptr != NULL; prev = ptr, ptr = ptr->next) {
224  if (item == ptr->item) {
225  prev->next = ptr->next;
226  if (prev->next == NULL) {
227  last = prev;
228  }
229  delete ptr;
230  numInList--;
231  break;
232  }
233  }
234  Assert(ptr != NULL); // should always find item!
235  }
236  Assert(!IsInList(item));
237 }

286 //----------------------------------------------------------------------
287 // SortedList::Insert
288 // Insert an "item" into a list, so that the list elements are
289 // sorted in increasing order.
290 //
291 // Allocate a cvLispListElement to keep track of the item.
292 // If the list is empty, then this will be the only element.
293 // Otherwise, walk through the list, one element at a time,
294 // to find where the new item should be placed.
295 //
296 // "item" is the thing to put on the list.
297 //----------------------------------------------------------------------
298 
299 template <class T>
300 void
301 cvSortedList<T>::Insert(T item)
302 {
303  cvLispListElement<T> *element = new cvLispListElement<T>(item);
304  cvLispListElement<T> *ptr; // keep track
305 
306  Assert(!IsInList(item));
307  if (this->IsEmpty()) { // if list is empty, put at front
308  this->first = element;
309  this->last = element;
310  } else if (compare(item, (this->first)->item) < 0) { // item goes at front
311  element->next = this->first;
312  this->first = element;
313  } else { // look for first elt in list bigger than item
314  for (ptr =this->first; ptr->next != NULL; ptr = ptr->next) {
315  if (compare(item, ptr->next->item) < 0) {
316  element->next = ptr->next;
317  ptr->next = element;
318  this->numInList++;
319  return;
320  }
321  }
322  (this->last)->next = element; // item goes at end of list
323  this->last = element;
324  }
325  this->numInList++;
326  Assert(IsInList(item));
327 }

329 //----------------------------------------------------------------------
330 // cvLispList::SanityCheck
331 // Test whether this is still a legal list.
332 //
333 // Tests: do I get to last starting from first?
334 // does the list have the right # of elements?
335 //----------------------------------------------------------------------
336 
337 template <class T>
338 void
339 cvLispList<T>::SanityCheck() const
340 {
341  cvLispListElement<T> *ptr;
342  int numFound;
343 
344  if (first == NULL) {
345  Assert((numInList == 0) && (last == NULL));
346  } else if (first == last) {
347  Assert((numInList == 1) && (last->next == NULL));
348  } else {
349  for (numFound = 1, ptr = first; ptr != last; ptr = ptr->next) {
350  numFound++;
351  Assert(numFound <= numInList); // prevent infinite loop
352  }
353  Assert(numFound == numInList);
354  Assert(last->next == NULL);
355  }
356 }
