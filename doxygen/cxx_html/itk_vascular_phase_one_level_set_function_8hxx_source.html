193 VascularPhaseOneLevelSetFunction< TImageType, TFeatureImageType >
194 ::ComputeUpdate(const NeighborhoodType & it, void *globalData,
195  const FloatOffsetType & offset){

209  VascularPhaseOneGlobalDataStruct *gdv = (VascularPhaseOneGlobalDataStruct *)globalData;
210 
211  PixelType sup_update = this->Superclass::ComputeUpdate(it,globalData,offset);
212 
213 
214  ScalarValueType tmp;
215  if ( this->GetCurvatureWeight() != ZERO )
216  {
217  //this invokes a new term in gd...
218  //(k-kc)*b
219  curvature_term = this->ComputeCurvatureTerm(it,offset,gdv);
220  curvature_term -= this->GetEquilibriumCurvature();
221  curvature_term *= this->GetCurvatureWeight();
222  tmp = this->CurvatureSpeed(it, offset,gd);
223 
224  this->ComputeProjectionTerm(it, offset,gd);
225  if(gdv->m_GradientProjection < ZERO){
226  tmp = (ScalarValueType)std::pow((double)tmp,(double)this->GetDecayModiferRatio());
227  }
228  //((k-kc)*b)*exp^rise
229  curvature_term *= tmp;
230 
231 
232  //gd->m_MaxCurvatureChange = vnl_math_max( gd->m_MaxCurvatureChange,
233  // vnl_math_abs(curvature_term) );
234 
235 
236  phi_gradient = ZERO;
237 
238  if ( curvature_term > ZERO )
239  {
240  for ( i = 0; i < ImageDimension; i++ )
241  {
242  phi_gradient += vnl_math_sqr( vnl_math_max(gd->m_dx_backward[i], ZERO) )
243  + vnl_math_sqr( vnl_math_min(gd->m_dx_forward[i], ZERO) );
244  }
245  }
246  else
247  {
248  for ( i = 0; i < ImageDimension; i++ )
249  {
250  phi_gradient += vnl_math_sqr( vnl_math_min(gd->m_dx_backward[i], ZERO) )
251  + vnl_math_sqr( vnl_math_max(gd->m_dx_forward[i], ZERO) );
252  }
253  }
254 
255 
256  // Collect energy change from propagation term. This will be used in
257  // calculating the maximum time step that can be taken for this iteration.
258  gd->m_MaxCurvatureChange =
259  vnl_math_max( gd->m_MaxCurvatureChange,
260  vnl_math_abs(curvature_term) );
261 
262  curvature_term *= vcl_sqrt(phi_gradient);
263 
264 
265 
266  }
267  else
268  {
269  curvature_term = ZERO;
270  tmp = ZERO;
271  }
272 
273  // Calculate the advection term.
274  // $\alpha \stackrel{\rightharpoonup}{F}(\mathbf{x})\cdot\nabla\phi $
275  //
276  // Here we can use a simple upwinding scheme since we know the
277  // sign of each directional component of the advective force.
278  //
279  if ( this->GetAdvectionWeight() != ZERO )
280  {
281  advection_field = this->AdvectionField(it, offset, gd);
282  advection_term = ZERO;
283 
284  for ( i = 0; i < ImageDimension; i++ )
285  {
286  x_energy = this->GetAdvectionWeight() * advection_field[i];
287 
288  if ( x_energy > ZERO )
289  {
290  advection_term += advection_field[i] * gd->m_dx_backward[i];
291  }
292  else
293  {
294  advection_term += advection_field[i] * gd->m_dx_forward[i];
295  }
296 
297  gd->m_MaxAdvectionChange =
298  vnl_math_max( gd->m_MaxAdvectionChange, vnl_math_abs(x_energy) );
299  }
300  advection_term *= this->GetAdvectionWeight();
301  }
302  else
303  {
304  advection_term = ZERO;
305  }
306 
307 
308  // Return the combination of all the terms.
309  PixelType update = -(PixelType)( -curvature_term
310  - advection_term);
311 
312  if(0){
313 // std::cout << "Speed: " << this->CurvatureSpeed(it, offset,gd) << std::endl;
314 // std::cout << "super update: " << sup_update << std::endl;
315  std::cout << update << " = (" << this->ComputeCurvatureTerm(it, offset, gd) << " - " << this->GetEquilibriumCurvature() << ") * " << this->GetCurvatureWeight() << " * " << tmp <<
316  " - " << (double)advection_term/(double)this->GetAdvectionWeight() << " * " << this->GetAdvectionWeight() << std::endl;
317 
318  }
319  this->GetCurrentCurvatureImage()->SetPixel(it.GetIndex(),curvature_term);
320  this->GetCurrentAdvectionImage()->SetPixel(it.GetIndex(),advection_term);
321  this->GetCurrentUpdateImage()->SetPixel(it.GetIndex(),update);
322 
323  return update;
