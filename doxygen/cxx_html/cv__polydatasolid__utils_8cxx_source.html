99 int PlyDtaUtils_GetFaceIds( vtkPolyData *geom, int *v_num_faces, int **v_faces)
100 {
101  //Initiate variables used by function
102  vtkSmartPointer<vtkIntArray> boundaryScalars;
103  vtkIdType faceId;
104  vtkIdType value;
105  vtkIdType max = 0;
106  int i;
107  int *faceNums;
108  bool *checkNums;
109  int check = 0;
110  int faceid = 0;
111  double range[2];
112  *v_faces = NULL;
113 
114  boundaryScalars = vtkSmartPointer<vtkIntArray>::New();
115  if (PlyDtaUtils_PDCheckArrayName(geom,1,"ModelFaceID") != CV_OK)
116  {
117  fprintf(stderr,"Array name 'ModelFaceID' does not exist. Regions must be identified \
118  and named 'ModelFaceID' prior to this function call");
119  return CV_ERROR;
120  }
121  boundaryScalars = vtkIntArray::SafeDownCast(geom->GetCellData()->GetArray("ModelFaceID"));
122 // boundaryScalars = static_cast<vtkIntArray*>(geom->GetCellData()->GetArray("ModelFaceID"));
123 
124  boundaryScalars->GetRange(range,0);
125 
126  max = range[1];
127 
128  checkNums = new bool[max];
129  for (i=0;i<max;i++)
130  {
131  checkNums[i] = false;
132  }
133 
134  for (faceId=0;faceId<geom->GetNumberOfPolys();faceId++)
135  {
136  value = boundaryScalars->GetValue(faceId);
137  if (checkNums[value-1] == false)
138  {
139  check++;
140  checkNums[value-1] = true;
141  }
142  }
143  *v_num_faces = check;
144 
145  faceNums = new int[check];
146  for (i=0;i<max;i++)
147  {
148  if (checkNums[i] == true)
149  {
150  faceNums[faceid++] = i+1;
151  }
152  }
153  *v_faces = faceNums;
154 
155  delete [] checkNums;
156 
157  return CV_OK;
158  //}
159 }

186

192  geom->GetCellData()->PassData(boundFacs->GetOutput()->GetCellData());
193  geom->BuildLinks();
194 
195  *numRegions = boundFacs->GetNumberOfRegions();
196 
197  return CV_OK;

286  }

291  reader->SetFileName(filename);
292  reader->Update();
293 
294  result->DeepCopy(reader->GetOutput());
295  result->BuildLinks();
296  }
297 
298  //Polygon File Format PLY
299  else if (!strncmp(extension,"ply",3)) {
300  vtkSmartPointer<vtkPLYReader> reader = vtkSmartPointer<vtkPLYReader>::New();
301  reader->SetFileName(filename);
302  reader->Update();
303

305  result->BuildLinks();
